# WSL2 Configuration for TRT Yocto Project
# Place this file as .wslconfig in your Windows user directory
# (C:\Users\<YourUsername>\.wslconfig)

[wsl2]
# Set memory limit (adjust based on your system)
memory=8GB

# Set processor limit (adjust based on your system)
processors=4

# Enable swap file

# Set swap file location

# Enable GUI applications (Windows 11)
guiApplications=true

# Network settings
networkingMode=mirrored
firewall=true

# Enable systemd (for Ubuntu 22.04+)
[ubuntu]
systemd=true
